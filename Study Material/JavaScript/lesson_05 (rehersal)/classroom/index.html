כתבו פונקצייה אשר מקבלת מספר, ומחזירה האם המספר הוא ראשוני או לא. 
function primeNumber(x){
    for(let i = 2; i < x; i++){
        if(x %i == 0){
            return `${x} is not a prime`;
        }
    }
    return `${x} is a prime`;
}
let isPrime = primeNumber(13);
console.log(isPrime);

כתבו פונקצייה שמקבלת מחרוזת, הפונקצייה תמיין את המחרוזת מהקטן לגדול. 
function reorder(x){
    let i,j;
    for(i = 0; i < x.length; i++){
        for(j = 0; j < x.length - i - 1; j++){
            if(x.charAt(j) > x.charAt(j + 1)){
                x = x.slice(0,j) + x.charAt(j+1) + x.charAt(j) + x.slice(j+2);
            }
        }
    }
    console.log(x);
}
